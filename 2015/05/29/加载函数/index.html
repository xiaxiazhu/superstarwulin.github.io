设计模式

加载器通用类

核心函数

/**
 *根据加载函数实现加载器
 *@param {Function} load 加载函数
 *@returns {Function} 加载器
 */
function createLoader(load) {
    var value, loading, handles = [], h;
    return function (handle, type) {
        //type默认为1
        //0:    不立即加载，不过加载完成之后执行回调函数
        //1:    立即加载，并在完成之后执行回调函数
        //2:    不立即加载，只在当前已经存在的情况下执行回调函数
        if (type !== 0 && !type) {
            type = 1;
        }
        if ((type & 1) && !loading) {
            loading = true;
            load(function (v) {
                value = v;
                while (h = handles.shift()) {
                    try {
                        h && h.apply(null, [value]);
                    }
                    catch (err) {
                        setTimeout(function () {
                            throw err;
                        }, 0)
                    }
                }
            })
        }
        if (value!==undefined) {
            handle && handle.apply(null, [value]);
            return value;
        }
        if (!(type & 2)) {
            handle && handles.push(handle);
        }//如果只在存在的情况下回调，则退出
        return value;
    }
}

调用方式

  1. 加载器实例onDatalazyload定义，createLoader定义为上文加载器通用类

    var onDatalazyload = createLoader(function(callback) {
    // 这里做网络请求或者数据与数据之间的联动触发，
    // 由callback(value)函数返回所获取的数据，
    // 下面是获取网络请求公共组件js文件，并且获取组件实例的例子

    S.use("mui/datalazyload", function (S, Datalazyload) {
    var dl = new Datalazyload(document.body, {
        autoDestroy: false,
        diff: {
            top: 0,
            bottom: 10
        }
    });
    callback(dl);
});

    })

  2. 所定义的加载器使用 — 执行器实例

    onDatalazyload(function(DL){

    // 以这样的方式获取定义的加载器所需要加载的内容
// 通过DL拿到调用实例
DL.addElements(self.$el.getDOMNode());
    self.page.on('hide', function(){
        DL.pause();
        }).on('show', function(){
            DL.resume();
    });

    });

优秀的特性以及代码解析

如果用这样的方式管理数据，或者管理远程的组件实例，就可以将管理对象集中起来，比如集中到一个文件道中，而调用实例，可以分布到各种各样的调用逻辑里面来执行。

执行器可以调用多次，加载器只发出一次网络请求，其他的从本地缓存获取。执行器如果没定义或者没执行，加载器不执行。

总结

对前端这样的富交互应用，大量的异步接口数据调用，上则提到加载器和调用器分离开来可以更好的维护代码以及分清模式，那一次加载，多次调用则让调用器更加广泛的应用到更多的逻辑中。

优秀特性中所提到的集中管理这样的微架构，将所有的数据集中起来管理，在大型网络富交互应用的前端部分可以让迭代开发更加容易，树形的模块结构显得更加可控，问题定位可以更快。
